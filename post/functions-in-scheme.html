Factorial and nCr() in Scheme

Recursive form, starting with $n!$:

(define factorial
    (lambda (n)
        (if (or (= n 0) (= n 1))
            1
            (* n (factorial (- n 1)))
)    )    )

nCr using $\displaystyle{ {n \choose r} = {n! \over r!(n-r)!} }$:

(define ncr
    (lambda (n r)
        (/ (factorial n) (* (factorial r) (factorial (- n r))))
)    )

and a safer version of nCr:

(define ncr
    (lambda (n r)
        (if (or (= n r) (= r 0))
            1
            (+ (ncr (- n 1) (- r 1)) (ncr (- n 1) r))
)    )    )
